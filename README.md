### Code I'm Proud Of

A small, dependency-free utility from the **Indian Amount Parser** — converts spoken Hindi, Bengali, Tamil and 17 other Indian language number words into an integer:

```js
// convert words like "एक लाख पच्चीस हज़ार" -> 125000
export function parseIndianAmount(input) {
  const text = normalize(input.toLowerCase());
  const tokens = text.split(/\s+/).filter(Boolean);

  const ONES  = { /* zero..nineteen across 20+ languages */ };
  const SCALE = { हज़ार: 1e3, लाख: 1e5, करोड़: 1e7, ... };

  let total = 0, group = 0;
  for (const t of tokens) {
    if (SCALE[t])          { group += SCALE[t]; continue; }
    if (t in ONES)         { group += ONES[t];  continue; }
    if (t === 'and' || t === 'और') continue;
  }
  return total + group;
}
```
